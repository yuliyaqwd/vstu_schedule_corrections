POST /api/apply-correction/

Применить корректировку к объекту

Пример запроса:

{
    "subject": {
        "value": "Программирование",
        "context": [
            {"key": "факультет", "value": "ИТ", "important": true}
        ]
    },
    "hypotheses": [
        {
            "value": "Программирование (Python)", 
            "score": 0.8
        }
    ],
    "scope_id": 1
}
